# LLM Arena Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build an LLM comparison tool where users type a Vue component prompt, run it against Claude Haiku, Sonnet, and Opus simultaneously, and compare response time, token usage, cost, generated code, and a live rendered preview.

**Architecture:** Frontend-orchestrated parallel requests — the Vue 3 frontend fires up to 3 simultaneous `POST /generate` calls (one per selected model), each card resolves independently as responses arrive. The Express backend is a thin pass-through to the Anthropic SDK. Live preview uses `vue3-sfc-loader` loaded via CDN inside a sandboxed `<iframe srcdoc>` — no backend changes needed for preview.

**Tech Stack:** Vue 3 + TypeScript + Tailwind CSS + Vite (frontend), Express + `@anthropic-ai/sdk` (backend), Monaco Editor (code display), `vue3-sfc-loader` CDN (live preview), Vitest + `@vue/test-utils` + `jsdom` (testing)

```js
import express from 'express'
import cors from 'cors'
import dotenv from 'dotenv'
import Anthropic from '@anthropic-ai/sdk'

dotenv.config()

const app = express()
app.use(cors())
app.use(express.json())

const client = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY })

const VALID_MODELS = new Set([
  'claude-haiku-4-5-20251001',
  'claude-sonnet-4-6',
  'claude-opus-4-7',
])

app.post('/generate', async (req, res) => {
  const { prompt, model } = req.body

  if (!prompt || !model || !VALID_MODELS.has(model)) {
    return res.status(400).json({ error: 'Invalid prompt or model' })
  }

  const start = Date.now()

  try {
    const response = await client.messages.create({
      model,
      max_tokens: 4096,
      system: 'You generate Vue 3 Single File Components using Tailwind CSS. Return only the component code, no markdown fences.',
      messages: [{ role: 'user', content: prompt }],
    })

    res.json({
      output: response.content[0].text,
      inputTokens: response.usage.input_tokens,
      outputTokens: response.usage.output_tokens,
      durationMs: Date.now() - start,
    })
  } catch (error) {
    console.error(error)
    res.status(500).json({ error: 'Generation failed' })
  }
})

app.listen(3000, () => console.log('Server running on port 3000'))
```

```ts
import { ref, readonly } from 'vue'
import axios from 'axios'

export type ModelId =
  | 'claude-haiku-4-5-20251001'
  | 'claude-sonnet-4-6'
  | 'claude-opus-4-7'

export type Status = 'idle' | 'loading' | 'done' | 'error'

const PRICING: Record<ModelId, { inputPerM: number; outputPerM: number }> = {
  'claude-haiku-4-5-20251001': { inputPerM: 0.80, outputPerM: 4.00 },
  'claude-sonnet-4-6':         { inputPerM: 3.00, outputPerM: 15.00 },
  'claude-opus-4-7':           { inputPerM: 15.00, outputPerM: 75.00 },
}

export function computeCost(model: ModelId, inputTokens: number, outputTokens: number): number {
  const p = PRICING[model]
  return (inputTokens / 1_000_000) * p.inputPerM + (outputTokens / 1_000_000) * p.outputPerM
}

export function useGenerate(model: ModelId) {
  const status = ref<Status>('idle')
  const output = ref('')
  const durationMs = ref(0)
  const inputTokens = ref(0)
  const outputTokens = ref(0)
  const cost = ref(0)

  async function run(prompt: string) {
    status.value = 'loading'
    output.value = ''
    durationMs.value = 0
    inputTokens.value = 0
    outputTokens.value = 0
    cost.value = 0

    try {
      const { data } = await axios.post('http://localhost:3000/generate', { prompt, model })
      output.value = data.output
      durationMs.value = data.durationMs
      inputTokens.value = data.inputTokens
      outputTokens.value = data.outputTokens
      cost.value = computeCost(model, data.inputTokens, data.outputTokens)
      status.value = 'done'
    } catch {
      status.value = 'error'
    }
  }

  return {
    status: readonly(status),
    output: readonly(output),
    durationMs: readonly(durationMs),
    inputTokens: readonly(inputTokens),
    outputTokens: readonly(outputTokens),
    cost: readonly(cost),
    run,
  }
}
```

```vue
<script setup lang="ts">
import { ref } from 'vue'
import type { ModelId } from '../composables/useGenerate'

const emit = defineEmits<{
  run: [models: ModelId[], prompt: string]
}>()

const ALL_MODELS: ModelId[] = [
  'claude-haiku-4-5-20251001',
  'claude-sonnet-4-6',
  'claude-opus-4-7',
]

const MODEL_LABELS: Record<ModelId, string> = {
  'claude-haiku-4-5-20251001': 'Haiku 4.5',
  'claude-sonnet-4-6': 'Sonnet 4.6',
  'claude-opus-4-7': 'Opus 4.7',
}

const prompt = ref('')
const selected = ref<Set<ModelId>>(new Set(ALL_MODELS))

function toggleModel(model: ModelId) {
  const next = new Set(selected.value)
  if (next.has(model)) next.delete(model)
  else next.add(model)
  selected.value = next
}

function runSelected() {
  const models = ALL_MODELS.filter(m => selected.value.has(m))
  if (prompt.value.trim() && models.length > 0) {
    emit('run', models, prompt.value.trim())
  }
}

function compareAll() {
  if (prompt.value.trim()) {
    emit('run', ALL_MODELS, prompt.value.trim())
  }
}
</script>

<template>
  <div class="flex flex-col gap-4">
    <textarea
      v-model="prompt"
      placeholder="Describe the component you want..."
      class="w-full h-32 p-3 border border-gray-300 rounded-lg resize-none focus:outline-none focus:ring-2 focus:ring-blue-500 font-sans text-sm"
    />
    <div class="flex items-center gap-6 flex-wrap">
      <div class="flex gap-4">
        <label
          v-for="model in ALL_MODELS"
          :key="model"
          class="flex items-center gap-1.5 cursor-pointer select-none text-sm font-medium text-gray-700"
        >
          <input
            type="checkbox"
            :checked="selected.has(model)"
            @change="toggleModel(model)"
            class="cursor-pointer accent-blue-500"
          />
          {{ MODEL_LABELS[model] }}
        </label>
      </div>
      <div class="flex gap-2 ml-auto">
        <button
          data-test="run-selected"
          @click="runSelected"
          :disabled="!prompt.trim() || selected.size === 0"
          class="px-4 py-2 bg-blue-500 text-white text-sm font-medium rounded-lg disabled:opacity-40 hover:bg-blue-600 transition-colors"
        >
          Run Selected
        </button>
        <button

          :disabled="!prompt.trim()"
          class="px-4 py-2 bg-purple-600 text-white text-sm font-medium rounded-lg disabled:opacity-40 hover:bg-purple-700 transition-colors"

          Compare All
        </button>
      </div>
    </div>
  </div>
</template>
```

```vue
<script setup lang="ts">
import { onMounted, watch, useTemplateRef } from 'vue'
import * as monaco from 'monaco-editor'

const props = defineProps<{ code: string }>()

const container = useTemplateRef<HTMLDivElement>('container')
let editor: monaco.editor.IStandaloneCodeEditor | null = null

onMounted(() => {
  if (!container.value) return
  editor = monaco.editor.create(container.value, {
    value: props.code,
    language: 'html',
    readOnly: true,
    theme: 'vs-dark',
    minimap: { enabled: false },
    scrollBeyondLastLine: false,
    fontSize: 13,
    lineNumbers: 'on',
    wordWrap: 'on',
  })
})

watch(() => props.code, newCode => {
  if (editor && editor.getValue() !== newCode) {
    editor.setValue(newCode)
  }
})
</script>

<template>
  <div ref="container" class="h-full w-full" />
</template>
```

The code is JSON-serialised inside the srcdoc so backticks, quotes, and special characters in the generated component cannot break the injected script.

```vue
<script setup lang="ts">
import { computed } from 'vue'

const props = defineProps<{ code: string }>()

const srcdoc = computed(() => {
  if (!props.code) return ''

  const encoded = JSON.stringify(props.code)

  return `<!DOCTYPE html>
<html>
<head>
  <meta charset="UTF-8" />
  <script src="https://cdn.tailwindcss.com"><\/script>
  <script src="https://unpkg.com/vue@3/dist/vue.global.js"><\/script>
  <script src="https://unpkg.com/vue3-sfc-loader/dist/vue3-sfc-loader.js"><\/script>
  <style>body { margin: 0; padding: 1rem; font-family: system-ui, sans-serif; }<\/style>
</head>
<body>
  <div id="app"></div>
  <div id="error" style="display:none;color:red;padding:1rem;font-family:monospace;font-size:12px;white-space:pre-wrap;"></div>
  <script>
    const sfcCode = ${encoded}
    const { loadModule } = window['vue3-sfc-loader']
    const options = {
      moduleCache: { vue: Vue },
      async getFile() { return sfcCode },
      addStyle(textContent) {
        const style = document.createElement('style')
        style.textContent = textContent
        document.head.appendChild(style)
      },
      handleError(err) {
        document.getElementById('error').style.display = 'block'
        document.getElementById('error').textContent = String(err)
      },
    }
    Vue.createApp(
      Vue.defineAsyncComponent(() => loadModule('component.vue', options))
    ).mount('#app')
  <\/script>
</body>
</html>`
})
</script>

<template>
  <div class="h-full w-full overflow-hidden">
    <iframe
      v-if="code"
      :srcdoc="srcdoc"
      sandbox="allow-scripts"
      class="w-full h-full border-0"
    />
    <div
      v-else
      class="flex items-center justify-center h-full text-gray-400 text-sm"
    >
      No output yet
    </div>
  </div>
</template>
```
